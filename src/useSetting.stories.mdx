# useSetting

```tsx
import {createConfigurableContext, createSetting, useSetting} from './index';

const ConfigurableContext = createConfigurableContext();
const MySetting = createSetting(ConfigurableContext);
const MyComponent = () => {
  const settingValue = useSetting(MySetting);

  return (<pre>MySetting is: {JSON.stringify(settingValue)}</pre>);
};
```
